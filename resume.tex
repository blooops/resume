%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SECTION EXPERIENCE %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Research Experience}

\vspace{4mm}

\cventry{2018--2019}{Identification of Human Host Factors for\textit{ Flaviviruses}}{}{}{}{
\textit{Senior Thesis (BTP)} at Indian Institute of Technology, \textit{Kharagpur, India}\\
\textit{Supervisor}: Prof. Riddhiman Dhar
\begin{itemize}[topsep=8pt,itemsep=4pt,parsep=0pt,partopsep=0pt]
    \item Analyzed genome wide screen data to identify proviral and antiviral host factors.
    \item Utilized non-parametric test (Mann-Whitney U test) with multiple testing correction.
    \item Determined relevant interaction pathways through gene ontology based networks.
\end{itemize}
}

\vspace{4mm}

\cventry{Summer 2018}{Fragment Based Modelling of ssRNA - Protein complexes}{}{}{}{
\textit{Summer Internship} at Inria Nancy (University of Lorraine), \textit{Nancy, France}\\
\textit{Supervisor}: Dr. Isaure Chauvot de Beauchene
\begin{itemize}[topsep=8pt,itemsep=4pt,parsep=0pt,partopsep=0pt]
    \item Extended the fragment based  \textcolor{blue}{\href{http://www.attract.ph.tum.de/services/ATTRACT/attract.html}{ATTRACT}} algorithm for ssRNA secondary structures.
    \item Achieved RMSD within 2 to 6 \AA \hspace{1pt} against known complexes: 4pkd, 2c4z, 1ec6, 5udz.
    \item Optimized the time complexity by parallelizing using numpy to enable scaling up.
\end{itemize}
}

\vspace{4mm}

\cventry{2017--2018}{Rigid Body Modelling of Protein - DNA Interactions}{}{}{}{
\textit{Research Project} at Indian Institute of Technology, \textit{Kharagpur, India}\\
\textit{Supervisor}: Prof. Agneyo Ganguly, Prof. Pralay Mitra
\begin{itemize}[topsep=8pt,itemsep=4pt,parsep=0pt,partopsep=0pt]
    \item Created a new docker using method of convolutions (Katchalski-Katzir algorithm). 
    \item Modelled 3-dimensional DNA strands of varying parameters using \textcolor{blue}{\href{https://x3dna.org/}{3DNA}}.
    \item Successful against the Haddock benchmark for easy and intermediate cases.
\end{itemize}
}
